PIN hashing (Node.js / bcrypt)
const bcrypt = require('bcrypt'); const ROUNDS = 12; // Hash on first setup async function hashPIN(pin) { return await bcrypt.hash(String(pin), ROUNDS); } // Verify on login async function verifyPIN(pin, hash) { return await bcrypt.compare(String(pin), hash); } // Login handler async function login(mobile, pin) { const worker = await db.workers .where('mobile', mobile).first(); if (!worker) throw 'Not found'; const ok = await verifyPIN(pin, worker.pin_hash); if (!ok) throw 'Invalid PIN'; return generateJWT(worker); }
